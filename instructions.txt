# Use an official Python image (here: Python 3.11 slim version for lighter image)
FROM python:3.11-slim

# Maintainer info and project label
LABEL maintainer="s10arnaud.wft@gmail.com"
LABEL project="Python_App"

# Set a working directory 
WORKDIR /app

# Copy project files into the image
COPY . /app

# Install Python dependencies 
RUN pip install --no-cache-dir 

# Set an environment variable
ENV APP_ENV=production

# Exposes port 5000
EXPOSE 5000

# Set user 
RUN useradd -m appuser
USER appuser

# Define the default command to run the app
CMD ["python", "app.py"]
